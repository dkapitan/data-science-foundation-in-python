```python
def Power3(x,a):
    result = x**a
    return result
```

## (e)


```python
def Plot(log=''):
    x = np.arange(1,10)
    y = Power3(x,2)
    
    #create plot
    fig, ax = plt.subplots()
    
    #config plot
    ax.set_xlabel('x')
    ax.set_ylabel('y=x^2')
    ax.set_title('Power3()')
    
    #change scale according to axis
    if log == 'x':
        ax.set_xscale('log')
        ax.set_xlabel('log(x)')
    if log == 'y':
        ax.set_yscale('log')
        ax.set_ylabel('log(y=x^2)')
    if log == 'xy':
        ax.set_xscale('log')
        ax.set_yscale('log')
        ax.set_xlabel('log(x)')
        ax.set_ylabel('log(y=x^2)')
    
    #draw plot
    ax.plot(x, y)
```


```python
Plot(log='xy')
```
